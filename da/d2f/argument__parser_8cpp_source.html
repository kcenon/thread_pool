181  std::map<std::string, std::string> argument_manager::parse(
182  const std::vector<std::string>& arguments)
183  {
184  std::map<std::string, std::string> result;
185 
186  size_t argc = arguments.size();
187  std::string argument_id;
188  for (size_t index = 0; index < argc; ++index)
189  {
190  argument_id = arguments[index];
191  size_t offset = argument_id.find("--", 0);
192  if (offset != 0)
193  {
194  continue;
195  }
196 
197  if (argument_id.compare("--help") == 0)
198  {
199  result.insert({ argument_id, "display help" });
200  continue;
201  }
202 
203  if (index + 1 >= argc)
204  {
205  break;
206  }
207 
208  auto target = result.find(argument_id);
209  if (target == result.end())
210  {
211  result.insert({ argument_id, arguments[index + 1] });
212  ++index;
213 
214  continue;
215  }
216 
217  target->second = arguments[index + 1];
218  ++index;
219  }
220 
221  return result;
222  }
