167  auto logger::start() -> std::tuple<bool, std::optional<std::string>>
168  {
169  if (collector_ == nullptr)
170  {
171  return { false, "there is no collector" };
172  }
173 
174  collector_->set_console_queue(console_writer_->get_job_queue());
175  collector_->set_file_queue(file_writer_->get_job_queue());
176 
177  auto [started, start_error] = collector_->start();
178  if (!started)
179  {
180  return { false, start_error };
181  }
182 
183  auto [console_started, console_start_error] = console_writer_->start();
184  if (!console_started)
185  {
186  return { false, console_start_error };
187  }
188 
189  auto [file_started, file_start_error] = file_writer_->start();
190  if (!file_started)
191  {
192  return { false, file_start_error };
193  }
194 
195  return { true, std::nullopt };
196  }
