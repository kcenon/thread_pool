118  auto log_job::do_work() -> std::tuple<bool, std::optional<std::string>>
119  {
120  try
121  {
122  std::string formatted_time =
123 #ifdef USE_STD_FORMAT
124  std::format
125 #else
126  fmt::format
127 #endif
128  ("{} {}.{}{}", datetime_tool::date(timestamp_), datetime_tool::time(timestamp_),
129  datetime_tool::milliseconds(timestamp_), datetime_tool::microseconds(timestamp_));
130 
131  if (start_time_.has_value())
132  {
133  auto time_gap = datetime_tool::time_difference<std::chrono::milliseconds,
134  std::chrono::high_resolution_clock>(
135  start_time_.value());
136 
137  if (!type_.has_value())
138  {
139  log_message_ =
140 #ifdef USE_STD_FORMAT
141  std::format
142 #else
143  fmt::format
144 #endif
145  ("[{}][{}] [{} ms]", formatted_time, convert_message(), time_gap);
146  }
147  else
148  {
149  log_message_ =
150 #ifdef USE_STD_FORMAT
151  std::format
152 #else
153  fmt::format
154 #endif
155  ("[{}][{}]: {} [{} ms]", formatted_time, type_.value(), convert_message(),
156  time_gap);
157  }
158 
159  return { true, std::nullopt };
160  }
161 
162  if (!type_.has_value())
163  {
164  log_message_ =
165 #ifdef USE_STD_FORMAT
166  std::format
167 #else
168  fmt::format
169 #endif
170  ("[{}][{}]", formatted_time, convert_message());
171  }
172  else
173  {
174  log_message_ =
175 #ifdef USE_STD_FORMAT
176  std::format
177 #else
178  fmt::format
179 #endif
180  ("[{}][{}]: {}", formatted_time, type_.value(), convert_message());
181  }
182 
183  return { true, std::nullopt };
184  }
185  catch (const std::exception& e)
186  {
187  return { false, std::string(e.what()) };
188  }
189  catch (...)
190  {
191  return { false, "unknown error" };
192  }
193  }

195  auto log_job::convert_message() const -> std::string
196  {
197  switch (message_type_)
198  {
199  case message_types::String:
200  {
201  return message_;
202  }
203  break;
204  case message_types::WString:
205  {
206  auto [converted, convert_error] = convert_string::to_string(wmessage_);
207  if (!converted.has_value())
208  {
209  return std::string();
210  }
211  return converted.value();
212  }
213  break;
214  case message_types::U16String:
215  {
216  auto [converted, convert_error] = convert_string::to_string(u16message_);
217  if (!converted.has_value())
218  {
219  return std::string();
220  }
221  return converted.value();
222  }
223  break;
224  case message_types::U32String:
225  {
226  auto [converted, convert_error] = convert_string::to_string(u32message_);
227  if (!converted.has_value())
228  {
229  return std::string();
230  }
231  return converted.value();
232  }
233  break;
234  default:
235  return std::string();
236  }
