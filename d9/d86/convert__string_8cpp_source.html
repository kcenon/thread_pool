40  auto convert_string::get_encoding_name(encoding_types encoding,
41  endian_types endian) -> std::string
42  {
43  switch (encoding)
44  {
45  case encoding_types::utf8:
46  return "UTF-8";
47  case encoding_types::utf16:
48  if (endian == endian_types::little)
49  return "UTF-16LE";
50  else if (endian == endian_types::big)
51  return "UTF-16BE";
52  else
53  return "UTF-16";
54  case encoding_types::utf32:
55  if (endian == endian_types::little)
56  return "UTF-32LE";
57  else if (endian == endian_types::big)
58  return "UTF-32BE";
59  else
60  return "UTF-32";
61  default:
62  throw std::runtime_error("Unknown encoding");
63  }
64  }

82  auto convert_string::detect_endian(const std::u16string& str) -> endian_types
83  {
84  if (str.empty())
85  return endian_types::unknown;
86 
87  if (str[0] == 0xFEFF)
88  return endian_types::big;
89  if (str[0] == 0xFFFE)
90  return endian_types::little;
91 
92  size_t sample_size = std::min<size_t>(str.size(), 1000);
93  int le_count = 0, be_count = 0;
94  for (size_t i = 0; i < sample_size; ++i)
95  {
96  uint16_t ch = str[i];
97  if ((ch & 0xFF00) == 0 && (ch & 0x00FF) != 0)
98  ++le_count;
99  if ((ch & 0x00FF) == 0 && (ch & 0xFF00) != 0)
100  ++be_count;
101  }
102 
103  if (le_count > be_count)
104  return endian_types::little;
105  if (be_count > le_count)
106  return endian_types::big;
107 
108  return endian_types::unknown;
109  }

111  auto convert_string::detect_endian(const std::u32string& str) -> endian_types
112  {
113  if (str.empty())
114  return endian_types::unknown;
115 
116  if (str[0] == 0x0000FEFF)
117  return endian_types::big;
118  if (str[0] == 0xFFFE0000)
119  return endian_types::little;
120 
121  size_t sample_size = std::min<size_t>(str.size(), 1000);
122  int le_count = 0, be_count = 0;
123  for (size_t i = 0; i < sample_size; ++i)
124  {
125  uint32_t ch = str[i];
126  if ((ch & 0xFFFFFF00) == 0 && (ch & 0x000000FF) != 0)
127  ++le_count;
128  if ((ch & 0x00FFFFFF) == 0 && (ch & 0xFF000000) != 0)
129  ++be_count;
130  }
131 
132  if (le_count > be_count)
133  return endian_types::little;
134  if (be_count > le_count)
135  return endian_types::big;
136 
137  return endian_types::unknown;

140  auto convert_string::has_utf8_bom(const std::string& str) -> bool
141  {
142  return str.length() >= 3 && static_cast<unsigned char>(str[0]) == 0xEF
143  && static_cast<unsigned char>(str[1]) == 0xBB

370  auto convert_string::split(const std::string& source, const std::string& token)
371  -> std::tuple<std::optional<std::vector<std::string>>, std::optional<std::string>>
372  {
373  if (token.empty())
374  {
375  return { std::nullopt, "Token cannot be an empty string" };
376  }
377 
378  std::vector<std::string> result;
379  size_t start = 0;
380  size_t endian = source.find(token);
381 
382  while (endian != std::string::npos)
383  {
384  result.emplace_back(source.substr(start, endian - start));
385  start = endian + token.length();
386  endian = source.find(token, start);
387  }
388 
389  result.emplace_back(source.substr(start));
390 
391  return { result, std::nullopt };
392  }
