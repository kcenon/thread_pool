210  template <typename FromType, typename ToType>
211  static auto convert(const FromType& value,
212  const std::string& from_encoding,
213  const std::string& to_encoding)
214  -> std::tuple<std::optional<ToType>, std::optional<std::string>>
215  {
216  iconv_t cd = iconv_open(to_encoding.c_str(), from_encoding.c_str());
217  if (cd == (iconv_t)-1)
218  {
219  return { std::nullopt, "iconv_open failed: " + std::string(strerror(errno)) };
220  }
221 
222  std::vector<char> in_buf = string_to_vector(value);
223  char* in_ptr = in_buf.data();
224  size_t in_bytes_left = in_buf.size();
225 
226  size_t out_buf_size = in_bytes_left * 2;
227  std::vector<char> out_buf(out_buf_size);
228  char* out_ptr = out_buf.data();
229  size_t out_bytes_left = out_buf.size();
230 
231  size_t result = iconv(cd, &in_ptr, &in_bytes_left, &out_ptr, &out_bytes_left);
232  while (result == (size_t)-1 && errno == E2BIG)
233  {
234  size_t used = out_buf_size - out_bytes_left;
235  out_buf_size *= 2;
236  out_buf.resize(out_buf_size);
237  out_ptr = out_buf.data() + used;
238  out_bytes_left = out_buf_size - used;
239  result = iconv(cd, &in_ptr, &in_bytes_left, &out_ptr, &out_bytes_left);
240  }
241 
242  if (result == (size_t)-1)
243  {
244  std::string error_msg = "iconv failed: " + std::string(strerror(errno));
245  iconv_close(cd);
246  return { std::nullopt, error_msg };
247  }
248 
249  iconv_close(cd);
250 
251  size_t converted_size = out_buf_size - out_bytes_left;
252  ToType converted_string(reinterpret_cast<typename ToType::value_type*>(out_buf.data()),
253  converted_size / sizeof(typename ToType::value_type));
254 
255  return { converted_string, std::nullopt };
256  }
