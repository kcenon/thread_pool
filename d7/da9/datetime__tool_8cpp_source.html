44  const std::string& separator_character) -> std::string
45  {
46 #ifdef USE_STD_CHRONO_CURRENT_ZONE

67  fmt::format

69  ("{:04d}{}{:02d}{}{:02d}", year, separator_character, month, separator_character, day);

72  auto datetime_tool::time(const std::chrono::system_clock::time_point& time,
73  const std::string& separator_character) -> std::string
74  {
75 #ifdef USE_STD_CHRONO_CURRENT_ZONE
76  const auto local_time = std::chrono::current_zone()->to_local(time);
77  const auto tod = std::chrono::hh_mm_ss{ std::chrono::floor<std::chrono::seconds>(
78  local_time - std::chrono::floor<std::chrono::days>(local_time)) };
79 
80  const auto hours = tod.hours().count();
81  const auto minutes = tod.minutes().count();
82  const auto seconds = tod.seconds().count();
83 #else
84  const auto time_t = std::chrono::system_clock::to_time_t(time);
85  const auto* tm = std::localtime(&time_t);
86 
87  const auto hours = tm->tm_hour;
88  const auto minutes = tm->tm_min;
89  const auto seconds = tm->tm_sec;
90 #endif
91 
92  return
93 #ifdef USE_STD_FORMAT
94  std::format
95 #else
96  fmt::format
97 #endif
98  ("{:02d}{}{:02d}{}{:02d}", hours, separator_character, minutes, separator_character,
99  seconds);
